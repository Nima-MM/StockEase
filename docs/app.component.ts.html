/**
 * The main application component.
 * 
 * This component sets up the main structure of the application, including the navbar, footer, ribbon, and login form.
 * It also provides the alert service to the rest of the application.
 * 
 * @component
 * @name App
 * 
 * @imports
 * - defineComponent: Function to define a Vue component.
 * - provide: Function to provide a dependency to the component.
 * - Ribbon: The ribbon component.
 * - JhiFooter: The footer component.
 * - JhiNavbar: The navbar component.
 * - LoginForm: The login form component.
 * - useAlertService: Function to use the alert service.
 * - '@/shared/config/dayjs': Configuration for dayjs.
 * 
 * @compatConfig {object} - Compatibility configuration for Vue 3.
 * @components {object} - The components used in this component.
 * @setup {function} - The setup function for the component.
 * 
 * @provides
 * - alertService: The alert service used for displaying alerts.
 */
import { defineComponent, provide } from 'vue';
import Ribbon from '@/core/ribbon/ribbon.vue';
import JhiFooter from '@/core/jhi-footer/jhi-footer.vue';
import JhiNavbar from '@/core/jhi-navbar/jhi-navbar.vue';
import LoginForm from '@/account/login-form/login-form.vue';

import { useAlertService } from '@/shared/alert/alert.service';

import '@/shared/config/dayjs';

export default defineComponent({
  
  name: 'App',
  components: {
    ribbon: Ribbon,
    'jhi-navbar': JhiNavbar,
    'login-form': LoginForm,
    'jhi-footer': JhiFooter,
  },
  setup() {
    provide('alertService', useAlertService());

    return {};
  },
});
